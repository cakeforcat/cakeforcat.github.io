  // initialize the digital pin as an output.
  pinMode(led, OUTPUT);     
}

void loop() {
  digitalWrite(led, HIGH);   // turn the LED on (HIGH is the voltage level)

  delay(2000);               // wait for a second
}

LED strip and light sensor

Here is a slightly more advanced challange if you want to try it out. We have a light sensor and an strip of 10 LEDs. These are smart LEDs and you can change their color and brightness individually. You need a special library called "Adafruit NeoPixel" to control them. Feel free to ask for some help getting it installed.

Here's and example code that will initialize the strip and play an animation on it. 

#include <Adafruit_NeoPixel.h>
#ifdef __AVR__
  #include <avr/power.h>
#endif
#define PIN        6
#define NUMPIXELS 16

Adafruit_NeoPixel pixels(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);
#define DELAYVAL 500

void setup() {
#if defined(__AVR_ATtiny85__) && (F_CPU == 16000000)
  clock_prescale_set(clock_div_1);
#endif

  pixels.begin();
}

void loop() {
  pixels.clear();

  for(int i=0; i<NUMPIXELS; i++) {

    pixels.setPixelColor(i, pixels.Color(0, 150, 0));
    pixels.show();
    delay(DELAYVAL);
  }
}

Ok, here's the challange! We have a few different sensors that will sense something in the room and convert that to a signal the Arduino can understand. One of them is a simple light sensor that will give a higher value when there's more light in the room.

There's a solution at the end but here are some useful code snippets to get you started: 

// Read the raw analog value from pin A0 (will be 0-1023)
int sensorValue = analogRead(A0);

// Map the sensor value (0-1023) to a brightness value (0-255)
int brightness = map(sensorValue, 0, 1023, 0, 255);

// Set the brightness of all pixels in the strip
for(int i=0; i<NUMPIXELS; i++) {
  pixels.setPixelColor(i, pixels.Color(brightness, brightness, brightness));
}
pixels.show();
