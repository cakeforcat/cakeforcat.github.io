Ada Lovelace Day 2025!

#include <Adafruit_NeoPixel.h>
#ifdef __AVR__
  #include <avr/power.h>
#endif

// Define the pin connected to the NeoPixel strip
#define PIN        10
// Define the number of pixels in the strip
#define NUMPIXELS 10
// Define the analog input pin
#define ANALOG_PIN A0

// Initialize the NeoPixel library
Adafruit_NeoPixel pixels(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);

void setup() {
  // This is for Trinket/Gemma/etc.
#if defined(__AVR_ATtiny85__) && (F_CPU == 16000000)
  clock_prescale_set(clock_div_1);
#endif

  pixels.begin(); // Initialize NeoPixel library
}

void loop() {
  // Read the raw analog value from pin A0 (will be 0-1023)
  int sensorValue = analogRead(ANALOG_PIN);
  
  // Map the analog value to the number of pixels.
  // The map() function re-scales the 0-1023 range to our 0-NUMPIXELS range.
  int ledsToLight = map(sensorValue, 0, 1023, 0, NUMPIXELS);

  pixels.clear(); // Set all pixel data to 'off'

  // This loop lights up the calculated number of LEDs
  for(int i = 0; i < ledsToLight; i++) {
    // Set pixel 'i' to a green color.
    // The Color() function takes Red, Green, Blue values (0-255)
    pixels.setPixelColor(i, pixels.Color(0, 150, 0));
  }

  pixels.show(); // Send the updated pixel data to the strip
  
  delay(50); // Add a small delay to prevent flickering
}
